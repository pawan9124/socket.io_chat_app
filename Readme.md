# Starting the app

# client side

1. cd client_ts
2. npm start
3. port running on 3000 or any

# backend side (local running/dev)

1. cd server_ts
2. npm run dev
3. port running on 4000

# backend side (build and start)

1. cd server_ts
2. npm run build
3. npm start

# nodejs and npm version
 1. nodejs = v14.17.1
 2. npm  = 6.14.13
